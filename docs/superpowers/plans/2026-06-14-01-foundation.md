#### Step 11: Implement lib/supabase/server.ts
```ts
import { createServerClient as createSSRServerClient } from "@supabase/ssr";
import { cookies } from "next/headers";
import type { Database } from "./types";

export async function createServerClient() {
  const cookieStore = await cookies();
  return createSSRServerClient<Database>(
    process.env.NEXT_PUBLIC_SUPABASE_URL!,
    process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!,
    {
      cookies: {
        getAll() {
          return cookieStore.getAll();
        },
        setAll(cookiesToSet) {
          try {
            cookiesToSet.forEach(({ name, value, options }) =>
              cookieStore.set(name, value, options)
            );
          } catch {
            // Called from Server Component, ignore
          }
        },
      },
    }
  );
}
```

#### Step 13: Implement lib/supabase/middleware.ts
```ts
import { createServerClient as createSSRServerClient } from "@supabase/ssr";
import { NextResponse, type NextRequest } from "next/server";
import type { Database } from "./types";

export async function updateSession(request: NextRequest) {
  let response = NextResponse.next({ request });

  const supabase = createSSRServerClient<Database>(
    process.env.NEXT_PUBLIC_SUPABASE_URL!,
    process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!,
    {
      cookies: {
        getAll() {
          return request.cookies.getAll();
        },
        setAll(cookiesToSet) {
          cookiesToSet.forEach(({ name, value }) =>
            request.cookies.set(name, value)
          );
          response = NextResponse.next({ request });
          cookiesToSet.forEach(({ name, value, options }) =>
            response.cookies.set(name, value, options)
          );
        },
      },
    }
  );

  // Refresh session if needed
  await supabase.auth.getUser();

  return response;
}
```

```sql
-- Atomic increment user usage with monthly limit check
CREATE OR REPLACE FUNCTION public.increment_user_usage(p_user_id UUID)
RETURNS TABLE(allowed BOOLEAN, remaining INT) AS $$
DECLARE
  v_count INT;
  v_limit INT := 50;
BEGIN
  -- Lock the row
  SELECT monthly_generation_count INTO v_count
  FROM public.user_profiles
  WHERE id = p_user_id
  FOR UPDATE;

  -- Reset if new month
  IF (SELECT date_trunc('month', monthly_reset_at) FROM public.user_profiles WHERE id = p_user_id)
     < date_trunc('month', now()) THEN
    UPDATE public.user_profiles
    SET monthly_generation_count = 1,
        monthly_reset_at = now()
    WHERE id = p_user_id;
    RETURN QUERY SELECT TRUE, v_limit - 1;
  ELSE
    IF v_count >= v_limit THEN
      RETURN QUERY SELECT FALSE, 0;
    ELSE
      UPDATE public.user_profiles
      SET monthly_generation_count = monthly_generation_count + 1
      WHERE id = p_user_id;
      RETURN QUERY SELECT TRUE, v_limit - (v_count + 1);
    END IF;
  END IF;
END;
$$ LANGUAGE plpgsql;
```

```ts
"use server";

import { signInSchema } from "@/lib/validation/auth";
import { createServerClient } from "@/lib/supabase/server";
import { headers } from "next/headers";
import { redirect } from "next/navigation";

export type AuthState = {
  error?: string;
  success?: boolean;
};

export async function signInWithMagicLink(
  _prev: AuthState,
  formData: FormData
): Promise<AuthState> {
  const parsed = signInSchema.safeParse({
    email: formData.get("email"),
  });

  if (!parsed.success) {
    return { error: parsed.error.issues[0].message };
  }

  const supabase = await createServerClient();
  const origin = (await headers()).get("origin");

  const { error } = await supabase.auth.signInWithOtp({
    email: parsed.data.email,
    options: {
      emailRedirectTo: `${origin}/api/auth/callback`,
    },
  });

  if (error) {
    return { error: "Gagal mengirim magic link. Coba lagi." };
  }

  return { success: true };
}

export async function signInWithGoogle(): Promise<void> {
  const supabase = await createServerClient();
  const origin = (await headers()).get("origin");

  const { data, error } = await supabase.auth.signInWithOAuth({
    provider: "google",
    options: {
      redirectTo: `${origin}/api/auth/callback`,
    },
  });

  if (error) {
    redirect("/login?error=oauth_failed");
  }

  if (data.url) {
    redirect(data.url);
  }
}

export async function signOut(): Promise<void> {
  const supabase = await createServerClient();
  await supabase.auth.signOut();
  redirect("/login");
}
```

```ts
import { NextResponse, type NextRequest } from "next/server";
import { createServerClient } from "@/lib/supabase/server";

export async function GET(request: NextRequest) {
  const { searchParams, origin } = new URL(request.url);
  const code = searchParams.get("code");
  const next = searchParams.get("next") ?? "/";

  if (code) {
    const supabase = await createServerClient();
    const { error } = await supabase.auth.exchangeCodeForSession(code);
    if (!error) {
      return NextResponse.redirect(`${origin}${next}`);
    }
  }

  return NextResponse.redirect(`${origin}/login?error=auth_failed`);
}
```

```ts
export interface RetryOptions {
  maxRetries?: number;
  baseDelay?: number;
}

export async function withRetry<T>(
  fn: () => Promise<T>,
  options: RetryOptions = {}
): Promise<T> {
  const { maxRetries = 3, baseDelay = 1000 } = options;
  let lastError: Error | null = null;

  for (let i = 0; i < maxRetries; i++) {
    try {
      return await fn();
    } catch (error: any) {
      lastError = error;
      const isLast = i === maxRetries - 1;
      const isRateLimit = error?.status === 429;

      if (isLast) throw lastError;

      const delay = isRateLimit
        ? (parseInt(error.headers?.["retry-after"] || "5") * 1000)
        : baseDelay * Math.pow(2, i);

      await new Promise((r) => setTimeout(r, delay));
    }
  }
  throw lastError!;
}
```

```ts
import OpenAI from "openai";
import { AI_CONFIG } from "./config";
import { withRetry } from "./retry";

export const DEFAULT_TEXT_MODEL = AI_CONFIG.primary.model;
export const REASONING_EFFORT = AI_CONFIG.primary.reasoning;

export const aiClient = new OpenAI({
  apiKey: AI_CONFIG.primary.apiKey,
  baseURL: AI_CONFIG.primary.baseURL,
});

export interface GenerateOptions {
  prompt: string;
  systemPrompt?: string;
  model?: string;
  reasoning?: "max" | "high" | "low" | "non-thinking";
  jsonMode?: boolean;
}

export interface GenerateResult {
  content: string;
  tokensUsed: number;
  durationMs: number;
}

export async function generateText(options: GenerateOptions): Promise<GenerateResult> {
  const startTime = Date.now();

  const result = await withRetry(async () => {
    return await aiClient.chat.completions.create({
      model: options.model || DEFAULT_TEXT_MODEL,
      messages: [
        ...(options.systemPrompt
          ? [{ role: "system" as const, content: options.systemPrompt }]
          : []),
        { role: "user" as const, content: options.prompt },
      ],
      ...(options.jsonMode ? { response_format: { type: "json_object" as const } } : {}),
    });
  });

  return {
    content: result.choices[0].message.content || "",
    tokensUsed: result.usage?.total_tokens || 0,
    durationMs: Date.now() - startTime,
  };
}
```

```ts
import { createServerClient } from "@/lib/supabase/server";

export interface UsageCheck {
  allowed: boolean;
  remaining: number;
}

export async function checkAndIncrementUsage(
  userId: string
): Promise<UsageCheck> {
  const supabase = await createServerClient();
  const { data, error } = await supabase.rpc("increment_user_usage", {
    p_user_id: userId,
  });

  if (error) {

  }

  return {
    allowed: data.allowed,
    remaining: data.remaining,
  };
}
```

```ts
"use server";

import { generateText } from "@/lib/ai/client";
import { analyzeImage } from "@/lib/image-analysis/blip";
import { PRODUCT_ANALYZE_PROMPT } from "@/lib/ai/prompts/product-analyze";
import { checkAndIncrementUsage } from "@/lib/usage/limits";
import { createServerClient } from "@/lib/supabase/server";
import { revalidatePath } from "next/cache";

export interface ProductAnalysis {
  name: string;
  category: string;
  brand: string;
  price: string;
  target_market: string;
  usp: string;
  benefits: string;
}

export type AnalysisState = {
  error?: string;
  data?: ProductAnalysis;
  success?: boolean;
};

export async function analyzeProduct(
  imageUrl: string,
  referenceLink: string
): Promise<AnalysisState> {
  try {
    const supabase = await createServerClient();
    const { data: { user } } = await supabase.auth.getUser();
    if (!user) {
      return { error: "Anda harus login terlebih dahulu" };
    }

    // Check usage
    const usage = await checkAndIncrementUsage(user.id);
    if (!usage.allowed) {
      return {
        error: `Anda sudah mencapai 50 generate bulan ini. Tunggu bulan depan.`,
      };
    }

    // Step 1: Analyze image with BLIP-2
    let imageDescription = "";
    if (imageUrl) {
      try {
        imageDescription = await analyzeImage(imageUrl);
      } catch (err) {
        console.error("BLIP-2 failed:", err);
        // Continue without image description
      }
    }

    // Step 2: Build prompt
    const linkContext = referenceLink || "Tidak ada link";
    const prompt = PRODUCT_ANALYZE_PROMPT(imageDescription, linkContext);

    // Step 3: Call AI
    const result = await generateText({
      prompt,
      jsonMode: true,
    });

    // Step 4: Parse JSON
    let analysis: ProductAnalysis;
    try {
      analysis = JSON.parse(result.content);
    } catch (err) {
      console.error("Failed to parse AI response:", result.content);
      return { error: "AI response tidak valid. Coba lagi." };
    }

    // Step 5: Save to product_analyses
    await supabase.from("product_analyses").insert({
      user_id: user.id,
      source_type: imageUrl && referenceLink ? "both" : imageUrl ? "image" : "link",
      source_url: imageUrl || referenceLink,
      analysis_result: analysis,
      tokens_used: result.tokensUsed,
    });

    return { data: analysis, success: true };
  } catch (err: any) {
    console.error("analyzeProduct error:", err);
    return { error: err.message || "Gagal menganalisis produk. Coba lagi." };
  }
}
```

```ts
  error?: string;
  success?: boolean;

export async function saveProduct(formData: FormData): Promise<SaveState> {
  try {
    const supabase = await createServerClient();
    const { data: { user } } = await supabase.auth.getUser();
    if (!user) {
      return { error: "Anda harus login terlebih dahulu" };
    }

    const raw = {
      name: formData.get("name") as string,
      category: formData.get("category") as string,
      brand: formData.get("brand") as string,
      price: formData.get("price") as string,
      target_market: formData.get("target_market") as string,
      usp: formData.get("usp") as string,
      benefits: formData.get("benefits") as string,
      image_url: formData.get("image_url") as string | null,
      reference_link: formData.get("reference_link") as string | null,
    };

    const parsed = productSchema.safeParse(raw);
    if (!parsed.success) {
      return { error: parsed.error.issues[0].message };
    }

    const { data, error } = await supabase
      .from("products")
      .insert({
        ...parsed.data,
        user_id: user.id,
      })
      .select()
      .single();

    if (error) {
      return { error: `Gagal menyimpan: ${error.message}` };
    }

    revalidatePath("/produk");
    return { success: true, productId: data.id };
  } catch (err: any) {
    return { error: err.message || "Gagal menyimpan produk" };
  }
}
```

```tsx
interface OnboardingModalProps {
  open: boolean;
  onClose: () => void;
}

export function OnboardingModal({ open, onClose }: OnboardingModalProps) {
  const [step, setStep] = useState(0);
  const [loading, setLoading] = useState(false);

  if (!open) return null;

  const handleComplete = async (loadSample: boolean) => {
    setLoading(true);
    await completeOnboarding(loadSample);
    setLoading(false);
    onClose();
  };

  return (
    <div className="fixed inset-0 bg-black/50 z-50 flex items-center justify-center p-4">
      <div className="bg-white rounded-2xl shadow-xl max-w-md w-full p-6 relative">
        <button
          onClick={onClose}
          className="absolute top-4 right-4 text-slate-400 hover:text-slate-600"
        >
          <X size={20} />
        </button>

        {step === 0 && (
          <div className="text-center">
            <div className="w-16 h-16 bg-indigo-600/10 text-indigo-600 rounded-2xl flex items-center justify-center mx-auto mb-4">
              <Sparkle size={32} weight="duotone" />
            </div>
            <h2 className="text-2xl font-bold text-slate-800 mb-2">
              Selamat Datang! 🎉
            </h2>
            <p className="text-sm text-slate-500 mb-6">
              AffiliateAI Studio membantu Anda membuat konten marketing 10x lebih cepat
              dengan AI. Mari kita mulai.
            </p>
            <Button onClick={() => setStep(1)} className="w-full">
              Lanjut
            </Button>
          </div>
        )}

        {step === 1 && (
          <div>
            <h2 className="text-xl font-bold text-slate-800 mb-2">
              Coba dengan Data Contoh?
            </h2>
            <p className="text-sm text-slate-500 mb-6">
              Kami akan menambahkan 2 produk contoh agar Anda bisa langsung
              mencoba semua fitur AI generator.
            </p>
            <div className="space-y-3">
              <Button
                onClick={() => handleComplete(true)}
                disabled={loading}
                className="w-full"
              >
                {loading ? "Memuat..." : "Ya, Tambahkan Data Contoh"}
              </Button>
              <Button
                onClick={() => handleComplete(false)}
                disabled={loading}
                variant="outline"
                className="w-full"
              >
                Mulai dari Kosong
              </Button>
            </div>
          </div>
        )}
      </div>
    </div>
  );
}
```
